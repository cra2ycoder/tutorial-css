Colors:

color: name

color: hex

color: rgb

color: rgba

HTML:

              
                  <section class="font-styles">
                    <p class="color name">color: text</p>
                    <p class="color hex">color: hex</p>
                    <p class="color rgb">color: rgb</p>
                    <p class="color rgba">color: rgba</p>
                  </section>

CSS:

            
                .color.name {
                  color: red;
                }
      
                .color.hex {
                  color: #ff00ff;
                }
      
                .color.rgb {
                  color: rgb(13, 195, 38);
                }
      
                .color.rgba {
                  color: rgba(0, 0, 0, 0.4);
                }
